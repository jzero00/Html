입력받아 배열에 저장

arr = new Array();
while(true){
    //입력
    input = prompt("이름 입력");
    //조건
    if(input == "" || input == null) break;
    //배열에 저장        
    arr.push(input);
    }
    document.getElementById("result1").innerHTML = arr;
}

살아온 날 구하기

today = new Date();
myday = new Date("1995.07.25");
    
time = today.getTime() - myday.getTime();
sec = time / 1000;
min = sec / 60;
hr = min / 60;
day = hr / 24;
yr = day / 365;
res = "new Date() : " + today.toLocaleDateString() + "
";
res += "new Date('1995.07.25') : " + myday.toLocaleDateString() + "
"
res += "살아온 초 : " + sec + "
";
res += "당신은 " + yr + "년 살았습니다.";
alert(time);
    
document.getElementById("result1").innerHTML = res;

앞으로의 날 구하기

        현재부터 500일 후의 날을 구한다
today = new Date();

// time = 현재의 getTime()  + 500일의 getTime()
millisec = today.getTime() + 500 * 24 * 60 * 60 * 1000;

// alert(millisec);


// millisec으로 새로운 Date 객체 생성
newday = new Date(millisec);

//출력
document.getElementById("result2").innerHTML = newday.toLocaleDateString();
